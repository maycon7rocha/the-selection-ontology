PREFIX selection: <http://www.iff.edu.br/schemas/selection#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
CONSTRUCT
{
    ?condition selection:class ?class .
    ?class a owl:Class ;
             owl:equivalentClass
             [
                a owl:Class ;
                owl:complementOf ?childConditionClass 
    		 ] .
}    
WHERE 
{ 
	?condition a selection:NotCondition .
    BIND(URI(CONCAT(STR(?condition),"Class")) AS ?class)
    
    ?condition selection:condition ?childCondition .
    BIND(URI(CONCAT(STR(?childCondition),"Class")) AS ?childConditionClass) 
    
}
